package problem2;

public class MyPriorityQueue implements PriorityQueue {

    Object data;
    int priority;
    MyPriorityQueue next;

  public MyPriorityQueue(Object data, int priority) {
    this.data = data;
    this.priority = priority;
  }


  @Override
  public void insert(Object d, int p) {
    MyPriorityQueue start = this;

    MyPriorityQueue temp = new MyPriorityQueue(d, p);
    if ((this).priority > p) {
      // Insert New Node before head
      temp.next = this;
      return;
    }
    else {
      while (start.next != null &&
          start.next.priority < p) {
        start = start.next;
      }
      temp.next = start.next;
      start.next = temp;
    }
    return;
  }

  @Override
  public Object remove() {
    MyPriorityQueue temp = this;
    temp.next = temp.next.next;
    return temp.data;
  }

  @Override
  public Object front() {
    return this.data;
  }

  @Override
  public boolean isEmpty() {
    return this == null;
  }
}
